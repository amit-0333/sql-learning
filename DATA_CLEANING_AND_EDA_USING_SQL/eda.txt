-- 2. NUMERICAL COLUMN ANALYSIS (Price)

-- 5 number summary
SELECT
    COUNT(Price)         AS count,
    MIN(Price)           AS min,
    MAX(Price)           AS max,
    ROUND(AVG(Price), 2) AS mean,
    ROUND(STD(Price), 2) AS std
FROM laptopdata;

-- Missing values
SELECT COUNT(*) AS missing_price FROM laptopdata WHERE Price IS NULL;

-- Outliers using IQR method
WITH ordered AS (
    SELECT Price, ROW_NUMBER() OVER (ORDER BY Price) AS rn,
    COUNT(*) OVER() AS total
    FROM laptopdata WHERE Price IS NOT NULL
),
quartiles AS (
    SELECT
        AVG(CASE WHEN rn BETWEEN FLOOR(total*0.25) AND CEIL(total*0.25) THEN Price END) AS q1,
        AVG(CASE WHEN rn BETWEEN FLOOR(total*0.75) AND CEIL(total*0.75) THEN Price END) AS q3
    FROM ordered
),
bounds AS (
    SELECT
        q1 - 1.5 * (q3 - q1) AS lower_bound,
        q3 + 1.5 * (q3 - q1) AS upper_bound
    FROM quartiles
)
SELECT * FROM laptopdata
WHERE Price < (SELECT lower_bound FROM bounds)
   OR Price > (SELECT upper_bound FROM bounds);

-- Horizontal histogram (price distribution in buckets)
SELECT
    CONCAT(FLOOR(Price/10000)*10000, ' - ', FLOOR(Price/10000)*10000+9999) AS price_range,
    COUNT(*)                                                                 AS count,
    RPAD('', COUNT(*)/5, '*')                                               AS bar
FROM laptopdata
GROUP BY FLOOR(Price/10000)
ORDER BY FLOOR(Price/10000);


-- 3. CATEGORICAL COLUMN ANALYSIS (Company)

-- Value counts with percentage
SELECT
    Company,
    COUNT(*)                                         AS count,
    ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 2) AS percentage
FROM laptopdata
GROUP BY Company
ORDER BY count DESC;

-- Missing values
SELECT COUNT(*) AS missing_company FROM laptopdata WHERE Company IS NULL;

-- Pie chart simulation (count per category)
SELECT
    Company,
    COUNT(*)              AS count,
    RPAD('', COUNT(*)/3, '*') AS bar
FROM laptopdata
GROUP BY Company
ORDER BY count DESC;


-- 4. NUMERICAL - NUMERICAL (Price vs Ram)

-- Side by side 8 number summary
SELECT
    'Price' AS metric,
    COUNT(Price)         AS count,
    MIN(Price)           AS min,
    MAX(Price)           AS max,
    ROUND(AVG(Price), 2) AS mean,
    ROUND(STD(Price), 2) AS std
FROM laptopdata
UNION ALL
SELECT
    'Ram' AS metric,
    COUNT(Ram)           AS count,
    MIN(Ram)             AS min,
    MAX(Ram)             AS max,
    ROUND(AVG(Ram), 2)   AS mean,
    ROUND(STD(Ram), 2)   AS std
FROM laptopdata;

-- Scatterplot simulation (avg price per ram bucket)
SELECT
    Ram,
    ROUND(AVG(Price), 2) AS avg_price,
    COUNT(*)             AS count,
    RPAD('', COUNT(*)/2, '*') AS bar
FROM laptopdata
GROUP BY Ram
ORDER BY Ram;

-- Correlation between Price and Ram
SELECT
    ROUND(
        (COUNT(*) * SUM(Price * Ram) - SUM(Price) * SUM(Ram)) /
        SQRT(
            (COUNT(*) * SUM(Price * Price) - SUM(Price) * SUM(Price)) *
            (COUNT(*) * SUM(Ram * Ram)     - SUM(Ram)   * SUM(Ram))
        ), 4
    ) AS price_ram_correlation
FROM laptopdata;


-- 5. CATEGORICAL - CATEGORICAL (Company vs Gpu)

-- Contingency table
SELECT
    Company,
    SUM(Gpu = 'Intel')  AS Intel,
    SUM(Gpu = 'Nvidia') AS Nvidia,
    SUM(Gpu = 'AMD')    AS AMD,
    SUM(Gpu = 'ARM')    AS ARM,
    SUM(Gpu = 'Other')  AS Other,
    COUNT(*)            AS total
FROM laptopdata
GROUP BY Company
ORDER BY total DESC;

-- Stacked bar chart simulation
SELECT
    Company,
    Gpu,
    COUNT(*) AS count,
    RPAD('', COUNT(*)/2, '*') AS bar
FROM laptopdata
GROUP BY Company, Gpu
ORDER BY Company, count DESC;


-- 6. NUMERICAL - CATEGORICAL (Price vs Company)

-- 8. MISSING VALUE TREATMENT

-- Count nulls per column
SELECT
    SUM(Company IS NULL)           AS Company_nulls,
    SUM(TypeName IS NULL)          AS TypeName_nulls,
    SUM(Inches IS NULL)            AS Inches_nulls,
    SUM(screen_type IS NULL)       AS screen_type_nulls,
    SUM(resolution_width IS NULL)  AS resolution_width_nulls,
    SUM(resolution_height IS NULL) AS resolution_height_nulls,
    SUM(touchscreen IS NULL)       AS touchscreen_nulls,
    SUM(cpu_brand IS NULL)         AS cpu_brand_nulls,
    SUM(cpu_model IS NULL)         AS cpu_model_nulls,
    SUM(cpu_speed IS NULL)         AS cpu_speed_nulls,
    SUM(Ram IS NULL)               AS Ram_nulls,
    SUM(Gpu IS NULL)               AS Gpu_nulls,
    SUM(storage_type IS NULL)      AS storage_type_nulls,
    SUM(primary_storage IS NULL)   AS primary_storage_nulls,
    SUM(secondary_storage IS NULL) AS secondary_storage_nulls,
    SUM(OpSys IS NULL)             AS OpSys_nulls,
    SUM(Weight IS NULL)            AS Weight_nulls,
    SUM(Price IS NULL)             AS Price_nulls
FROM laptopdata;

-- Fill Weight nulls with median (numerical → median)
UPDATE laptopdata
SET Weight = (
    SELECT AVG(w) FROM (
        SELECT Weight AS w FROM laptopdata
        WHERE Weight IS NOT NULL
        ORDER BY Weight
        LIMIT 2 - (SELECT COUNT(*) FROM laptopdata WHERE Weight IS NOT NULL) % 2
        OFFSET (SELECT (COUNT(*) - 1) / 2 FROM laptopdata WHERE Weight IS NOT NULL)
    ) AS median_calc
)
WHERE Weight IS NULL;

-- Fill storage_type nulls with mode (categorical → mode)
UPDATE laptopdata
SET storage_type = (
    SELECT storage_type FROM laptopdata
    WHERE storage_type IS NOT NULL
    GROUP BY storage_type
    ORDER BY COUNT(*) DESC
    LIMIT 1
)
WHERE storage_type IS NULL;


-- 9. FEATURE ENGINEERING

-- Add ppi column (pixels per inch)
ALTER TABLE laptopdata ADD COLUMN ppi INT;

UPDATE laptopdata
SET ppi = ROUND(
    SQRT(resolution_width * resolution_width + resolution_height * resolution_height) / Inches
);

-- Add price_bracket column
ALTER TABLE laptopdata ADD COLUMN price_bracket VARCHAR(10);

UPDATE laptopdata
SET price_bracket = CASE
    WHEN Price < 25000  THEN 'Budget'
    WHEN Price < 50000  THEN 'Mid'
    WHEN Price < 100000 THEN 'High'
    ELSE 'Premium'
END;

-- Verify new features
SELECT `index`, Inches, resolution_width, resolution_height, ppi, Price, price_bracket
FROM laptopdata LIMIT 10;
